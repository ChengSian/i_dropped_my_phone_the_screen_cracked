Distortion

An overview of available distortion effects

Waveshaper

//create & connect a sampler to overdrive to output //crank the drive to 100 __(). sampler({path:"../../data/various/gun-cock.wav",loop:true}). waveshaper(100). dac(.25); __("sampler").start();

Overdrive

//create & connect a sampler to overdrive to output //crank the drive to 100 __(). sampler({path:"../../data/various/gun-cock.wav",loop:true}). overdrive(100). dac(.25); __("sampler").start();

//create & connect a sampler to output //trebly overdrive __(). sampler({path:"../../data/various/gun-cock.wav",loop:true}). overdrive({drive:50,color:5000,postCut:8000}). dac(.25); __("sampler").start();

Ring Modulator

//create & connect a sampler to output //ring modulator - default settings - frequency : 30, distortion : 1 __(). sampler({path:"../../data/various/gun-cock.wav",loop:true}). ring(). dac(.1); __("sampler").start();

//create a sampler and connect to output //ring modulator - default settings - frequency : 30, distortion : 1 __(). sampler({path:"../../data/various/gun-cock.wav",loop:true}). ring({frequency:2000,distortion:50}). dac(.025); __("sampler").start();

Bit Crusher

//create & connect a sampler to output //ring modulator - default settings - frequency : 30, distortion : 1 __(). sampler({path:"../../data/various/gun-cock.wav",loop:true}). bitcrusher(). dac(.5); __("sampler").start();

//decimate __(). sampler({path:"../../data/various/gun-cock.wav",loop:true}). bitcrusher({bits:2,frequency:.01}). dac(); __("sampler").start();
